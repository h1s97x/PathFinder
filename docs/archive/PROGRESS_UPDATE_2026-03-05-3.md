### 核心代码
```typescript
interface HistoryState {
  past: GraphJSON[]
  future: GraphJSON[]
}

// 撤销
undo: () => {
  const previous = past[past.length - 1]
  return {
    graph: Graph.fromJSON(previous),
    history: {
      past: past.slice(0, -1),
      future: [current, ...future],
    },
  }
}

// 重做
redo: () => {
  const next = future[0]
  return {
    graph: Graph.fromJSON(next),
    history: {
      past: [...past, current],
      future: future.slice(1),
    },
  }
}
```
